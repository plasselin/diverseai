import { Injectable } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { Observable } from 'rxjs';
import { tap } from 'rxjs/operators';
import { RegisterRequest, LoginRequest } from './auth-request.model';
import { RegisterResponse, LoginResponse } from './auth-response.model';
import { User } from './user.model';
import { JwtHelperService } from '@auth0/angular-jwt';
import { CookieService } from 'ngx-cookie-service';

/**
 * @service
 * @description
 * The AuthService manages authentication-related operations, including user registration,
 * login, email confirmation, and token management via cookies. It provides methods to
 * interact with the backend API and maintain authentication state.
 */
@Injectable({
    providedIn: 'root'
})
export class AuthService {
    /**
     * @property {string} apiUrl
     * @description
     * The base URL for authentication-related API endpoints.
     */
    private apiUrl = 'http://localhost:8000/api/auth/';

    /**
     * @property {string} hostUrl
     * @description
     * The base URL for the frontend application. Adjust as per your deployment.
     */
    private hostUrl = 'http://localhost:4200/';

    /**
     * @property {JwtHelperService} jwtHelper
     * @description
     * Service used to decode and validate JWT tokens.
     */
    private jwtHelper = new JwtHelperService();

    constructor(
        private http: HttpClient,
        private cookieService: CookieService
    ) {}

    /**
     * @method
     * @name register
     * @description
     * Registers a new user by sending the registration data to the backend API.
     * @param {RegisterRequest} data - The registration data containing user details.
     * @returns {Observable<RegisterResponse>} - An observable emitting the registration response.
     */
    register(data: RegisterRequest): Observable<RegisterResponse> {
        return this.http.post<RegisterResponse>(`${this.apiUrl}register/`, data);
    }

    /**
     * @method
     * @name login
     * @description
     * Logs in a user by sending the login credentials to the backend API.
     * @param {LoginRequest} data - The login credentials containing email and password.
     * @returns {Observable<LoginResponse>} - An observable emitting the login response (tokens).
     */
    login(data: LoginRequest): Observable<LoginResponse> {
        return this.http.post<LoginResponse>(`${this.apiUrl}login/`, data).pipe(
            tap(response => {
                this.setToken('access_token', response.access);
                this.setToken('refresh_token', response.refresh);
            })
        );
    }

    /**
     * @method
     * @name confirmEmail
     * @description
     * Confirms a user's email by sending the user ID and token to the backend API.
     * @param {string} uidb64 - Encoded user ID.
     * @param {string} token - Confirmation token.
     * @returns {Observable<any>} - An observable emitting the confirmation response.
     */
    confirmEmail(uidb64: string, token: string): Observable<any> {
        return this.http.get(`${this.apiUrl}confirm-email/${uidb64}/${token}/`);
    }

    /**
     * @method
     * @name resendConfirmationEmail
     * @description
     * Requests the backend API to resend the email confirmation link to the user's email.
     * @param {string} email - The user's email address.
     * @returns {Observable<any>} - An observable emitting the resend confirmation response.
     */
    resendConfirmationEmail(email: string): Observable<any> {
        return this.http.post(`${this.apiUrl}resend-confirmation/`, { email });
    }

    /**
     * @method
     * @name logout
     * @description
     * Logs out the user by removing authentication tokens from cookies.
     */
    logout(): void {
        this.removeToken('access_token');
        this.removeToken('refresh_token');
    }

    /**
     * @method
     * @name setToken
     * @description
     * Stores a token in cookies with a 24-hour expiration.
     * @param {string} tokenName - The name of the cookie (e.g., 'access_token').
     * @param {string} token - The JWT token to store.
     */
    setToken(tokenName: string, token: string): void {
        // Parameters: name, value, daysToExpire, path, domain, secure, sameSite
        this.cookieService.set(tokenName, token, 1, '/', 'yourdomain.com', true, 'Strict');
        /**
         * @note
         * Replace 'yourdomain.com' with your actual domain to ensure cookies are set correctly.
         * Consider configuring the domain dynamically based on the environment.
         */
    }

    /**
     * @method
     * @name getToken
     * @description
     * Retrieves a token from cookies.
     * @param {string} tokenName - The name of the cookie to retrieve.
     * @returns {string | null} - The token string if present, otherwise null.
     */
    getToken(tokenName: string): string | null {
        const token = this.cookieService.get(tokenName);
        return token ? token : null;
    }

    /**
     * @method
     * @name removeToken
     * @description
     * Removes a token from cookies.
     * @param {string} tokenName - The name of the cookie to remove.
     */
    removeToken(tokenName: string): void {
        this.cookieService.delete(tokenName, '/');
    }

    /**
     * @method
     * @name getDecodedToken
     * @description
     * Decodes a JWT token to extract its payload.
     * @param {string} tokenName - The name of the cookie storing the token.
     * @returns {any} - The decoded token payload if token exists, otherwise null.
     */
    getDecodedToken(tokenName: string): any {
        const token = this.getToken(tokenName);
        if (token) {
            return this.jwtHelper.decodeToken(token);
        }
        return null;
    }

    /**
     * @method
     * @name isTokenExpired
     * @description
     * Checks if a token is expired using JwtHelperService.
     * @param {string} tokenName - The name of the cookie storing the token.
     * @returns {boolean} - True if the token is expired or not present, false otherwise.
     */
    isTokenExpired(tokenName: string): boolean {
        const token = this.getToken(tokenName);
        if (token) {
            return this.jwtHelper.isTokenExpired(token);
        }
        return true;
    }

    /**
     * @method
     * @name isAuthenticated
     * @description
     * Determines if the user is authenticated by checking the validity of the access token.
     * @returns {boolean} - True if authenticated, false otherwise.
     */
    isAuthenticated(): boolean {
        const token = this.getToken('access_token');
        return token != null && !this.jwtHelper.isTokenExpired(token);
    }
}
