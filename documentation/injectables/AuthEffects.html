import { Injectable } from '@angular/core';
import { Actions, createEffect, ofType } from '@ngrx/effects';
import * as AuthActions from './auth.actions';
import { AuthService } from './auth.service';
import { catchError, map, mergeMap, tap } from 'rxjs/operators';
import { of } from 'rxjs';
import { Router } from '@angular/router';
import { JwtHelperService } from '@auth0/angular-jwt';

/**
 * @class
 * @description
 * AuthEffects handles side effects for authentication actions, such as making API calls,
 * storing tokens, and navigating the user based on action outcomes.
 */
@Injectable()
export class AuthEffects {

    constructor(
        private actions$: Actions,
        private authService: AuthService,
        private router: Router
    ) {}

    /**
     * @effect
     * @description
     * Listens for the 'register' action, invokes the AuthService's register method,
     * and dispatches either 'registerSuccess' or 'registerFailure' based on the API response.
     */
    register$ = createEffect(() =>
        this.actions$.pipe(
            ofType(AuthActions.register),
            mergeMap(action =>
                this.authService.register(action.registerData).pipe(
                    map(response => AuthActions.registerSuccess({ response })),
                    catchError(error => of(AuthActions.registerFailure({ error })))
                )
            )
        )
    );

    /**
     * @effect
     * @description
     * Upon successful registration, navigates the user to the email confirmation page.
     * Does not dispatch a new action.
     */
    registerSuccess$ = createEffect(() =>
            this.actions$.pipe(
                ofType(AuthActions.registerSuccess),
                tap(() => {
                    this.router.navigate(['/auth/email-confirmation']);
                })
            ),
        { dispatch: false }
    );

    /**
     * @effect
     * @description
     * Handles registration failures by logging the error. Further error handling
     * (e.g., displaying notifications) can be implemented here.
     * Does not dispatch a new action.
     */
    registerFailure$ = createEffect(() =>
            this.actions$.pipe(
                ofType(AuthActions.registerFailure),
                tap(action => {
                    console.error('Registration failed:', action.error);
                    // TODO: Implement user-friendly error notifications
                })
            ),
        { dispatch: false }
    );

    /**
     * @effect
     * @description
     * Listens for the 'login' action, invokes the AuthService's login method,
     * and dispatches either 'loginSuccess' or 'loginFailure' based on the API response.
     */
    login$ = createEffect(() =>
        this.actions$.pipe(
            ofType(AuthActions.login),
            mergeMap(action =>
                this.authService.login(action.loginData).pipe(
                    map(response => AuthActions.loginSuccess({ response })),
                    catchError(error => of(AuthActions.loginFailure({ error })))
                )
            )
        )
    );

    /**
     * @effect
     * @description
     * Upon successful login, navigates the user to the home page.
     * Does not dispatch a new action.
     */
    loginSuccess$ = createEffect(() =>
            this.actions$.pipe(
                ofType(AuthActions.loginSuccess),
                tap(() => {
                    this.router.navigate(['/home']);
                })
            ),
        { dispatch: false }
    );

    /**
     * @effect
     * @description
     * Handles login failures by logging the error. Further error handling
     * (e.g., displaying notifications) can be implemented here.
     * Does not dispatch a new action.
     */
    loginFailure$ = createEffect(() =>
            this.actions$.pipe(
                ofType(AuthActions.loginFailure),
                tap(action => {
                    console.error('Login failed:', action.error);
                    // TODO: Implement user-friendly error notifications
                })
            ),
        { dispatch: false }
    );

    /**
     * @effect
     * @description
     * Listens for the 'logout' action, invokes the AuthService's logout method,
     * and navigates the user to the login page.
     * Does not dispatch a new action.
     */
    logout$ = createEffect(() =>
            this.actions$.pipe(
                ofType(AuthActions.logout),
                tap(() => {
                    this.authService.logout();
                    this.router.navigate(['/auth/login']);
                })
            ),
        { dispatch: false }
    );

    /**
     * @effect
     * @description
     * Listens for the 'confirmEmail' action, invokes the AuthService's confirmEmail method,
     * and dispatches either 'confirmEmailSuccess' or 'confirmEmailFailure' based on the API response.
     */
    confirmEmail$ = createEffect(() =>
        this.actions$.pipe(
            ofType(AuthActions.confirmEmail),
            mergeMap(action =>
                this.authService.confirmEmail(action.uidb64, action.token).pipe(
                    map(response => AuthActions.confirmEmailSuccess({ message: response.message })),
                    catchError(error => of(AuthActions.confirmEmailFailure({ error })))
                )
            )
        )
    );

    /**
     * @effect
     * @description
     * Upon successful email confirmation, navigates the user to the login page.
     * Does not dispatch a new action.
     */
    confirmEmailSuccess$ = createEffect(() =>
            this.actions$.pipe(
                ofType(AuthActions.confirmEmailSuccess),
                tap(() => {
                    this.router.navigate(['/auth/login']);
                })
            ),
        { dispatch: false }
    );

    /**
     * @effect
     * @description
     * Handles email confirmation failures by logging the error. Further error handling
     * (e.g., displaying notifications) can be implemented here.
     * Does not dispatch a new action.
     */
    confirmEmailFailure$ = createEffect(() =>
            this.actions$.pipe(
                ofType(AuthActions.confirmEmailFailure),
                tap(action => {
                    console.error('Email confirmation failed:', action.error);
                    // TODO: Implement user-friendly error notifications
                })
            ),
        { dispatch: false }
    );
}
